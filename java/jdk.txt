# manual

tar xvzf jdk-8u161-linux-x64.tar.gz
sudo mkdir -p /opt/jdk1.8.0_161
sudo chown -R root:root /opt/jdk1.8.0_161/
sudo cp -r jdk1.8.0_161/* /opt/jdk1.8.0_161/

sudo rm /usr/lib/jvm/default-java
sudo ln -f -s /opt/jdk1.8.0_161 /usr/lib/jvm/default-java

sudo bash -c "cat > /etc/profile.d/java.sh" <<"EOF"
export JAVA_HOME=/usr/lib/jvm/default-java
export PATH=$JAVA_HOME/bin:$PATH
export CLASSPATH=$CLASSPATH:$JAVA_HOME/lib/
export JAVA_OPTS="-Xmx16g"
EOF

source /etc/profile.d/java.sh
